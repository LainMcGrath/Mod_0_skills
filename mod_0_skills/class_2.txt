class DiningRoom

seatMaterial(array)
numberOfTables(integer)
numberOfEmptyTables(integer)
tableType(array)


changeSeatMaterial(changes seatMaterial type - e.g. pleather to leather)
increaseNumberOfTables(adds tables to the restaurant)
calculateOpenDiningCapacity(divides numberOfEmptyTables by numberOfTables)
changeTableType (changes table type from 4 top to booth)
